        // 模拟已上传文件数据（实际应从后端API获取）
        const mockFiles = [

            },
            {

                type: 'image'
            },
            {
                id: 3,
                name: '胃镜报告_王某_20260625.pdf',
                size: '3.8 MB',
                uploadTime: '2026-06-25 16:50:22',
                type: 'pdf'
            }
        ];

        function renderFileList(files) {
            const container = document.getElementById('fileTableContainer');
            const countSpan = document.getElementById('fileCount');
            
            if (!files || files.length === 0) {
                container.innerHTML = `
                    <div class="empty-state">
                        <div class="icon">📭</div>
                        <p>暂无已上传的病例文件</p>
                    </div>
                `;
                countSpan.textContent = '共 0 个文件';
                return;
            }

            countSpan.textContent = `共 ${files.length} 个文件`;
            
            let html = `
                <table>
                    <thead>
                        <tr>
                            <th>文件名</th>
                            <th>类型</th>
                            <th>大小</th>
                            <th>上传时间</th>
                        </tr>
                    </thead>
                    <tbody>
            `;
            
            files.forEach(file => {
                const icon = file.type === 'pdf' ? '📄' : '🖼️';
                const badgeText = file.type === 'pdf' ? 'PDF' : '图片';
                html += `
                    <tr>
                        <td>
                            <span class="file-name">
                                <span class="file-icon">${icon}</span>
                                ${file.name}
                            </span>
                        </td>
                        <td><span class="badge">${badgeText}</span></td>
                        <td>${file.size}</td>
                        <td>${file.uploadTime}</td>
                    </tr>
                `;
            });

            html += '</tbody></table>';
            container.innerHTML = html;
        }

        function loadFileList() {
            // 实际部署时，此处应发起 fetch 请求到后端 API
            // 例如：fetch('/api/files').then(res => res.json()).then(data => renderFileList(data));
            
            // 目前使用模拟数据演示
            renderFileList(mockFiles);
        }

        // 页面加载时自动获取文件列表
        window.addEventListener('DOMContentLoaded', () => {
            loadFileList();
        });
    </script>
